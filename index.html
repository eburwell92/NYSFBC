Marching Band Scores Over Time

packages = ["pandas", "plotly"] import pandas as pd import plotly.express as px from js import document from pyodide.ffi import create_proxy from io import StringIO def handle_file(event): file = event.target.files[0] if not file: return # Read CSV text = file.text().to_py() df = pd.read_csv(StringIO(text)) # Parse dates df["Date"] = pd.to_datetime(df["Date"]) # Create interactive line chart fig = px.line(df, x="Date", y="Score", title="Marching Band Scores Over Time") fig.update_traces(mode="lines+markers") # Display chart in div document.getElementById("chart").innerHTML = fig.to_html(include_plotlyjs="cdn", full_html=False) # Connect file input to handler document.getElementById("file-input").addEventListener("change", create_proxy(handle_file))
